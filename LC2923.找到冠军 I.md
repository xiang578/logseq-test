## 参考代码

```cpp
class Solution {

	public:
		int findChampion(vector<vector<int>>& grid) {
		
		int n = grid.size();
		int ans = 0;
		
		for (int i=0;i<n;++i) {
		
			int sum = 0;
			for (int j=0;j<n;++j) {
				sum+=grid[i][j];
			}
          
			if (sum == n-1) {
				ans = i;
			}
		}
		return ans;
	}

};
```
