class Solution {
public:
    string smallestString(string s) {
        int ok = 0;
        for (int i=0;i<s.size();i++) {

            if (s[i] != 'a' && ok == 0) {
                ok = 1;
            }
            // cout<<s[i]<<endl;
            if (ok {{< logseq/mark >}} 1 && s[i] {{< / logseq/mark >}} 'a') {
                break;
            }
            if(ok == 1) s[i] = s[i] - 1;

        }
        if (ok == 0) s[s.size()-1] = 'z';
       return s; 
    }
};
