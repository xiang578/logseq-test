## 题解
简单动态规划
## 参考代码
```cpp
class Solution {
public:
    int minFallingPathSum(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        int dp[n][m];
        int ans = 1e9;
        memset(dp, 0, sizeof dp);
        for (int i = 0;i<n;i++) {
            for(int j = 0; j<m; j++) {
                if (i!=0) {
                    dp[i][j] = dp[i-1][j];
                    if (j > 0) dp[i][j] = min(dp[i][j], dp[i-1][j-1]);
                    if (j + 1 < m) dp[i][j] = min(dp[i][j], dp[i-1][j+1]);  
                }
                dp[i][j] += a[i][j];
                if (i==n-1) {
                    ans = min(ans, dp[i][j]);
                }
            }
        }
        return ans;
    }
};
```
