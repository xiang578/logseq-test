// @lc code=start
class MyStack {
    vector<int>q;
public:

    MyStack() {
    }
    
    void push(int x) {
        q.push_back(x);
    }
    
    int pop() {
        int ret = q[q.size()-1];
        q.pop_back();
        return ret;
    }
    
    int top() {
        return q[q.size()-1];
    }
    
    bool empty() {
        if (q.size() == 0) return true;
        return false;
    }
};
