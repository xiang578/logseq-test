## 参考代码
```cpp
class Solution {
public:
    int kItemsWithMaximumSum(int numOnes, int numZeros, int numNegOnes, int k) {
        int ans=0;
        if (k>=numOnes) {
            ans += numOnes;
            k-=numOnes;
        } else {
            ans = k;
            k = 0;
        }
        if(k>=numZeros) {
            k-=numZeros;
        } else {
            k = 0;
        }
        ans -= k;
        return ans;
    }
};
```
