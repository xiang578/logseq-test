## 题解

  + 位运算，按题意模拟。

## 参考代码

```cpp
/*
 * @lc app=leetcode.cn id=2917 lang=cpp
 *
 * [2917] 找出数组中的 K-or 值
 */

// @lc code=start
class Solution {
public:
    int findKOr(vector<int>& nums, int k) {
        long long ans = 0;
        int a[32];
        memset(a, 0, sizeof a);
        for(auto num:nums) {
            for(int i=0; i<31; i++) {
                if(num & (1<<i)) a[i]+=1;
            }
        }
        for(int i=0; i<31;i++) {
            if (a[i] >= k) {
                ans += (1<<i);
            }
        }
        return ans; 
    }
};
// @lc code=end


```
